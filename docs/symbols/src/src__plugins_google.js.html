  7 function replaceAll( str, from, to ) {
  8 	var idx;
  9 	try { idx = str.search( from );
 10 	} catch(e) { idx = str.toString().indexOf(from)}
 11 	while ( idx > -1 ) {
 12 		str = str.replace( from, to );
 13 		idx = str.indexOf( from );
 14 	}
 15 	return str;
 16 }

 18 core.unregisterPlugin("google");
 19 core.registerPlugin(Event.MESSAGE, function(bot, event, args, priv) {
 20 	var msg = args[args.length-1];
 21 	var googleURL = "http://www.google.com/search?q=";
 22 	if(msg.indexOf("google ") > -1) {
 23 		var query = msg.substring(7,msg.length);
 24 		query = replaceAll(query, " ","+");
 25 		var results = IO.fetchURL(googleURL+query, false);
 26 		var search_for = "<h3 class=\"r\"><a href=";
 27 		results = results.substring(results.indexOf(search_for)+search_for.length,results.length);
 28 		url=results.substring(1,results.indexOf('"',2));
 29 		results = results.substring(results.indexOf(')">')+3,results.indexOf("</a>")).replace(/(<([^>]+)>)/ig,"");
 30 		bot.sendMessage(args[0], results+" URL: "+url);
 31 		return true;
 32 	}
 33 }, "google");
