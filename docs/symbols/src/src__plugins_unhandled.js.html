1 /**********************************
  2 * A simple plugin which points    *
  3 * out of a message wasn't handled *
  4 * handled by the bot.             *
  5 **********************************/
  6 
  7 core.registerUnhandledEventPlugin(Event.MESSAGE, function(bot, event, args, priv) {
  8 	var elipsis = (args[args.length-1].length > 20)? "..." : "";
  9 	bot.sendMessage(args[0], "I don't know anything about '" + args[args.length-1].substring(0, Math.min(args[args.length-1].length, 20)) + elipsis + "', " + args[1]);
 10 });
