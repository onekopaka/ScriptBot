1 /**************************************************
  2 * A test plugin to reload specified files. The    *
  3 * filename given is automatically assumed to be   *
  4 * ".js", so the extension should be omitted. In   *
  5 * order for this plugin to work fully (and not    *
  6 * just load a plugin twice) the specified plugin  *
  7 * must have a line to unregister any previous     *
  8 * copies of the plugin. This can be done by       *
  9 * attaching an id to teh plugin and removing it   *
 10 * by id in the first line of the file (which will *
 11 * do nothing if it isn't loaded.                  *
 12 **************************************************/
 13 
 14 // Reloads the specified plugin.
 15 core.registerPlugin(Event.MESSAGE, function(bot, event, args, priv) {
 16 	var msg = args[args.length-1].toLowerCase();
 17 
 18 	// Only respond to "time" messages.
 19 	if(msg.substring(0, 6) === "reload") {
 20 		msg = Util.trim(msg.substring(6));
 21 
 22 		if(msg !== "" && msg !== "reload") {
 23 			try {
 24 				IO.include("plugins" + IO.slash + msg + ".js");
 25 				bot.sendMessage(args[0], args[1] + ": Success!");
 26 			} catch(e) {
 27 				bot.sendMessage(args[0], "Reloading plugin failed. Check your spelling, or maybe the plugin doesn't exist.");
 28 			}
 29 		} else {
 30 			bot.sendMessage(args[0], "Please specify a plugin name to be reloaded, " + args[1]);
 31 		}
 32 
 33 		return true;
 34 	}
 35 });
 36 core.registerPluginInfo("reload", function(bot, event, args, priv) {
 37 	bot.sendMessage(args[0], bot.prefix + "reload [plugin] - Used for testing, (re)loads the given plugin's file.");
 38 });
