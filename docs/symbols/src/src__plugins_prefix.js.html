1 /************************************************
  2 * A simple plugin to allow changing the prefix  *
  3 * the bot responds to in the channel. This does *
  4 * not affect unrestricted plugins.              *
  5 ************************************************/
  6 
  7 // Unregister old plugins.
  8 core.unregisterPluginByEvent(Event.MESSAGE, "prefix");
  9 // Allows users to change the prefix in channel. Mainly just because or testing.
 10 core.registerPlugin(Event.MESSAGE, function(bot, event, args, priv) {
 11 	var msg = args[args.length-1].split(/\s+/g);
 12 	if(msg[0] === "prefix" && msg.length === 1) {
 13 		bot.sendMessage(args[0], "I can be addressed using '" + bot.prefix + "', " + args[1]);
 14 		return true;
 15 	} else if(msg[0] === "prefix" && msg.length > 1) {
 16 		bot.prefix = msg[1];
 17 		bot.sendMessage(args[0], "I will now respond to '" + bot.prefix + "', " + args[1]);
 18 		return true;
 19 	}
 20 }, "prefix");
 21 core.registerPluginInfo("prefix", function(bot, event, args, priv) {
 22 	bot.sendMessage(args[0], bot.prefix + "prefix [new] - Displays the current prefix for me. If an optional parameter is given, it will set the current prefix to it.");
 23 });
