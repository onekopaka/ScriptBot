1 /**
  2  * A collection of useful functions to handle various things or convert Java objects to JS equivalents.
  3  *
  4  * @author AMcBain, 2008
  5  * @since 2.0.1
  6  * @namespace
  7  */
  8 var Util = {
  9 
 10 	/**
 11 	 * Checks to see if a Java object is a Java boolean with the value "true",
 12 	 * and returns a JavaScript boolean literal.Only guaranteed to work
 13 	 * correctly for Java Boolean objects.
 14 	 *
 15 	 * @since 2.0.1
 16 	 * @param {java.lang.Boolean} value The object to be checked.
 17 	 * @returns a JavaScript boolean literal.
 18 	 * @type boolean
 19 	 */
 20 	isTrue: function(value) {
 21 		return (value+"" === "true");
 22 	},
 23 
 24 	/**
 25 	 * Checks to see if a Java object represents the JavaScript value
 26 	 * of undefined and returns a JavaScript boolean literal.Only
 27 	 * guaranteed to work correctly for Java undefined objects.
 28 	 *
 29 	 * @since 2.0.1
 30 	 * @param {Object} value The object to be checked.
 31 	 * @returns a JavaScript boolean literal.
 32 	 * @type boolean
 33 	 */
 34 	isUndefined: function(value) {
 35 		return (value+"" === "undefined");
 36 	},
 37 
 38 	/**
 39 	 * Removes white-space characters from the beginning of the given value.
 40 	 *
 41 	 * @since 2.0.1
 42 	 * @param {string} value The string to be trimmed.
 43 	 * @returns a string
 44 	 * @type string
 45 	 */
 46 	trim: function(value) {
 47 		return (value+"").replace(/^\s+/,"").replace(/\s+$/,"");
 48 
 49 	},
 50 
 51 	/**
 52 	 * Converts a Java String-Object to a JavaScript string literal.
 53 	 *
 54 	 * @since 2.0.1
 55 	 * @param {java.lang.String} value The object to be converted.
 56 	 * @returns a JavaScript string literal.
 57 	 * @type string
 58 	 */
 59 	toJsString: function(value) {
 60 		return (value+"");
 61 	},
 62 
 63 	/**
 64 	 * Verifies the submitted message is intended for the specified target.
 65 	 *
 66 	 * @since 2.0.1
 67 	 * @param {string} target The target to be checked for.
 68 	 * @param {string} message The message to be checked.
 69 	 * @returns <code>true</code> if the message is for the target, <code>false</code> otherwise.
 70 	 */
 71 	verifyTarget: function(target, message) {
 72 		return ((message+"").match(new RegExp("^" + target + "\\W?\\s?")) != null);
 73 	},
 74 	
 75 	/**
 76 	 * Strips HTML-like tags from a string.
 77 	 *
 78 	 * @since 2.0.4
 79 	 * @param {string} dirtystring The string to strip of HTML-like tags.
 80 	 * @returns a string clean of all HTML-like tags.
 81 	 */
 82 	stripHTML: function(dirtystring) {
 83 		return dirtystring.replace(/<\/?[^>]+(>|$)/g,"");
 84 	}
 85 }
