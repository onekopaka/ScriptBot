1 /**
  2  * Gets contents of the specified URL. An optional parameter determines the format of the returned data.
  3  *
  4  * @since 2.0.1
  5  * @param {string} url The url/location to be fetched.
  6  * @param {boolean} [asLines] <code>true</code> returns the contents as a \n (new line) separated string, <code>false</code> to return the contents as an array of lines.
  7  * @returns A string or an Array containing the contents of the requested URL.
  8  */
  9 IO.fetchURL = function(url, asLines) {
 10 	try {
 11 
 12 		System.setProperty("http.agent", "Mozilla/5.0 (Windows NT 6.1; WOW64; rv:13.0) Gecko/20120415 Firefox/13.0a2");
 13 		var link = new java.net.URL(url);
 14 		var conn = link.openConnection();
 15 		var reader = new java.io.BufferedReader(new java.io.InputStreamReader(conn.getInputStream()));
 16 		var input = "";
 17 
 18 		var line = "";
 19 		var lines = [];
 20 
 21 		while((input = reader.readLine()) != null) {
 22 			if(asLines) {
 23 				lines.push(input+"");
 24 			} else {
 25 				line += input + "\n";
 26 			}
 27 		}
 28 
 29 		try {
 30 			reader.close();
 31 		} catch(e) {
 32 			// Uhh ... what am I supposed to do with it?
 33 		}
 34 
 35 		return ((asLines)? lines : line);
 36 	} catch(e) {
 37 		throw (new Error(e+""));
 38 	}
 39 };

 41 /**
 42  * Reads in the specified file. Optionally returns the contents as a string or an array of lines.
 43  *
 44  * @since 2.0.1
 45  * @param {string} path The path to the file.
 46  * @param {boolean} asLines <code>true</code> if the output should be returned as an array, <code>false</code> if as a string.
 47  * @returns <code>true</code> if the output should be returned as an array, <code>false</code> if as a string.
 48  */
 49 IO.readFile = function(path, asLines) {
 50 	var error = null;
 51 
 52 	// Store the file temporarily.
 53 	var lines = [];
 54 	var out = "";
 55 	var line = "";
 56 
 57 	// Create a new file reader.
 58 	var input = null;
 59 
 60 	try {
 61 		input = new java.io.BufferedReader(new java.io.FileReader(new java.io.File(path)));
 62 
 63 		do {
 64 			// Read in the line from the file. 
 65 			line = input.readLine();
 66 
 67 			if(line != null) {
 68 				// Add the line to our list.
 69 				if(asLines) {
 70 					lines.push(line);
 71 				} else {
 72 					out += line + "\n";
 73 				}
 74 			}
 75 		} while(line != null);
 76 	} catch (e) {
 77 		print("There was an error reading the file.");
 78 		error = e.toString() + "";
 79 	} finally {
 80 		try {
 81 			// Try to close the file. It's a good idea and practice.
 82 			input.close();
 83 		} catch (E) {
 84 			// We couldn't even close the file!?
 85 			print("Cannot close file. There is no file to close.");
 86 		}
 87 	}
 88 
 89 	if(error !== null) {
 90 		throw (new Error(error));
 91 	}
 92 
 93 	return ((asLines)? lines : out);
 94 };

 96 /**
 97  * Writes the given contents to the specified path.
 98  *
 99  * @since 2.0.1
100  * @param {string} path Path to the file to be written.
101  * @param {string} contents The contents of the file to be written.
102  * @param {boolean} [append] Whether to append the given contents instead of overwriting.
103  */
104 IO.writeFile = function(path, contents, append) {
105 	var contents = contents;
106 	if(!(contents instanceof Array)) {
107 		contents = [contents];
108 	}
109 
110 	// Append file contents?
111 	var apnd = (append)? true : false; // Make it a real boolean for Java.
112 
113 	var out = null;
114 	var error = null;
115 	try {
116 		out = new java.io.PrintWriter(new java.io.BufferedWriter(new java.io.FileWriter(path, apnd)));
117 		//Write summary file. 
118 		for (var i = 0; i < contents.length; i++) {
119 			out.println(contents[i]);
120 		}
121 	} catch(e) {
122 		print("There was an error writing the file");
123 		error = e.toString() + "";
124 	} finally {
125 		try {
126 			out.close();
127 		} catch (E) {
128 			print("Cannot close file. There is no file to close.");
129 		}
130 	}
131 
132 	if(error !== null) {
133 		throw (new Error(error));
134 	}
135 };

169 /**
170  * Writes the given object out to a file such that it can be read in via {@link #readObject}.
171  * Similar to a "pretty printer" which formats an object so as to be easily viewed in the output window.
172  * Note that all primivites will be wrapped in an array for output.
173  *
174  * @since 2.0.1
175  * @param {string} name The name of the object to be saved, to be used when read back in.
176  * @param {object} object The object to be saved.
177  * @param {Function} [funct] A function used to modify the output. Consult JSON documentation.
178  * @param {string} [tabin] Used to specify the indentation used in output.
179  */
180 IO.writeObject = function(name, object, funct, tabin) {
181 
182 	// Create the data directory if it exists.
183 	if(!Util.isTrue((new java.io.File(IO.path + "plugins/data/")).exists())) {
184 		(new java.io.File(IO.path + "plugins/data/")).mkdir();
185 	}
186 
187 	// Wrap non-objects, such as primitives.
188 	if(!(object instanceof Object)) {
189 		object = [object];
190 	}
191 
192 	// Create and write the object.
193 	var output = "{}";
194 	try {
195 		output = JSON.stringify(object);
196 	} catch(e) {
197 		// We know.
198 	}
199 	IO.writeFile(IO.path + "plugins/data/" + name + ".js", output);
200 	return (output);
201 };

215 /**
216  * Lists the files and folders of a directory into an array
217  *
218  * @since 2.0.3
219  * @param {string} path The folder or path to read
220  */
221 IO.listDir = function(path) {
222 	path = path.replace("..", "");
223 	try {
224 		path = IO.path+path;
225 		var dir = new java.io.File(path);
226 		var children = dir.list();
227 		if (children != null) {
228 			for (i=0; i<children.length; i++) {
229 				child = new java.io.File(children[i]);
230 				if(child.isDirectory()) {
231 					children[i] = children[i]+"/";
232 				}
233 			}
234 		}
235 		return children;
236 	} catch(e) {
237 		print("Error path \""+path+"\" does not exist or cannot be read from");
238 	}
239 };
