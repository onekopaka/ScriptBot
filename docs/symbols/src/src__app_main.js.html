1 // Help message.
  2 print("");
  3 print("*NIX    - press ^C to force the program to end");
  4 print("Windows - press ^C, then type 'y' to force end");
  5 print("IRC     - use the quit command to close nicely\n");
  6 
  7 // Create a new PircBot instance and start it up.
  8 var core = createBot();
  9 
 10 // Get the config file contents.
 11 var config = JSON.parse(IO.readFile("config.txt"));
 12 print("You are currently running a bot named \""+config.name+"\".\n");
 13 print("Please visit http://scriptbot.theoks.net/ for the latest version");
 14 print("of ScriptBot.\n");
 15 core.password = config.password || "";
 16 
 17 // Load any plugins before we join IRC
 18 if(config.plugins) {
 19 	for(var i = 0; i < config.plugins.length; i++) {
 20 		try {
 21 			IO.include("plugins/" + config.plugins[i] + ".js");
 22 			print("'" + config.plugins[i] + "' loaded");
 23 		} catch(e) {
 24 			print("'" + config.plugins[i] + "' failed to load with this error:");
 25 			print(e);
 26 		}
 27 	}
 28 }
 29 
 30 // Connecting to the server sometimes fails.
 31 try {
 32 	if(!config.name) throw "Cannot connect to IRC, no bot name specified. Check config file.";
 33 	if(!config.server) throw "Cannot connect to IRC, no server specified. Check config file.";
 34 	if(!config.channels) throw "Cannot connect to IRC, no channels listed. Check config file.";
 35 	if(config.channels.length === 0) throw "Must specify at least one channel to join IRC. Check config file.";
 36 	if(config.prefix) core.prefix = config.prefix;
 37 	
 38 	// Join IRC.
 39 	core.initialize(config.name, config.server, config.channels[0], config.port);
 40 	
 41 	// Identify with NickServ
 42 	core.getBot().sendMessage("NickServ", "IDENTIFY "+config.nickServPass)
 43 	
 44 	// Join any specified channels.
 45 	var keys = config.keys || [];
 46 	for(var i = 1; i < config.channels.length; i++) {
 47 		core.getBot().joinChannel(config.channels[i], keys[i] || "");
 48 	}
 49 } catch(e) {
 50 	print(e);
 51 	print("Program terminated. You May Close Your Window.");
 52 	pause;
 53 }
