1 /**
  2  * A collection of useful functions to handle various things or convert Java objects to JS equivalents.
  3  *
  4  * @namespace
  5  */
  6 var Util = {
  7 
  8 	/**
  9 	 * Checks to see if a Java object is a Java boolean with the value "true",
 10 	 * and returns a JavaScript boolean literal.Only guaranteed to work
 11 	 * correctly for Java Boolean objects.
 12 	 *
 13 	 * @param {java.lang.Boolean} value The object to be checked.
 14 	 * @returns a JavaScript boolean literal.
 15 	 * @type boolean
 16 	 */
 17 	isTrue: function(value) {
 18 		return (value+"" === "true");
 19 	},
 20 
 21 	/**
 22 	 * Checks to see if a Java object represents the JavaScript value
 23 	 * of undefined and returns a JavaScript boolean literal.Only
 24 	 * guaranteed to work correctly for Java undefined objects.
 25 	 *
 26 	 * @param {Object} value The object to be checked.
 27 	 * @returns a JavaScript boolean literal.
 28 	 * @type boolean
 29 	 */
 30 	isUndefined: function(value) {
 31 		return (value+"" === "undefined");
 32 	},
 33 
 34 	/**
 35 	 * Converts a Java String-Object to a JavaScript string literal.
 36 	 *
 37 	 * @param {java.lang.String} value The object to be converted.
 38 	 * @returns a JavaScript string literal.
 39 	 * @type string
 40 	 */
 41 	toJsString: function(value) {
 42 		return (value+"");
 43 	},
 44 
 45 	/**
 46 	 * Verifies the submitted message is intended for the specified target.
 47 	 *
 48 	 * @param {string} target The target to be checked for.
 49 	 * @param {string} message The message to be checked.
 50 	 * @returns <code>true</code> if the message for the target, <code>false</code> otherwise.
 51 	 */
 52 	verifyTarget: function(target, message) {
 53 		return ((message+"").match(new RegExp("^" + target + "\\W?\\s?")) != null);
 54 	}
 55 }
