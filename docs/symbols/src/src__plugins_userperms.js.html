  6 core.unregisterPlugin("userperms");
  7 var perms = {};
  8 if(IO.objectExists("perms")) 
  9 {
 10 	perms = IO.readObject("perms");
 11 }
 12 core.registerPlugin(Event.MESSAGE, function(bot, event, args, priv) {
 13 	var msg = args[args.length-1];
 14 	//give perms
 15 	if(msg.search("giveperms ") > -1 && args[1] != config.name)
 16 	{
 17 		if(perms[args[1]] !=1)
 18 		{
 19 			bot.sendMessage(args[0], "You don't have the permissions to edit user permissions");
 20 		}
 21 		else
 22 		{
 23 			username = msg.substring(msg.indexOf("giveperms ")+10, msg.length);
 24 			perms[username] = 1;
 25 			IO.writeObject("perms", perms);
 26 			bot.sendMessage(args[0], "The user " +username + " has been given permissions.");
 27 		}
 28 	}
 29 	//no perms for username!!
 30 	else if(msg.search("removeperms ") > -1 && args[1] != config.name)
 31 	{
 32 		if(perms[args[1]] !=1)
 33 		{
 34 			bot.sendMessage(args[0], "You don't have the permissions to edit user permissions");
 35 		}
 36 		else
 37 		{
 38 			username = msg.substring(msg.indexOf("giveperms ")+10, msg.length);
 39 			perms[username] = 0;
 40 			IO.writeObject("perms", perms);
 41 			bot.sendMessage(args[0], username+ "'s permissions were removed.");
 42 		}
 43 	}
 44 	//show perms
 45 	if(msg.search("showperms") > -1)
 46 	{
 47 		try
 48 		{
 49 			username = msg.substring(msg.indexOf("showperms ")+10, msg.length);
 50 			if(perms[args[1]] !=1)
 51 			{
 52 				bot.sendMessage(args[0], "The user " +username+ " has permissions.");
 53 			}
 54 		}
 55 		catch(e)
 56 		{
 57 			bot.sendMessage(args[0], "The specified user does not exist or does not have permissions");
 58 		}
 59 	}
 60 	return true;
 61 }, "userperms");
 62 
 63 core.registerPluginInfo("troutsaver", function(bot, event, args, priv) {
 64 	bot.sendMessage(args[0], bot.prefix + "Restricts some users from edtiing the brain.js object");
 65 });
