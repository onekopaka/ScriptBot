1 importClass(java.lang.System);
  2 
  3 /**
  4  * A collection of items related to helping with input/output, as well as loading other script files.
  5  *
  6  * @author AMcBain, 2008
  7  * @since 2.0.1
  8  * @namespace
  9  */
 10 var IO = {
 11 
 12 	/**
 13 	 * The line separator used by the system.
 14 	 * @string
 15 	 */
 16 	lineBreak: System.getProperty("line.separator") || "\n",
 17 
 18 	/**
 19 	 * The path separator used by the system.
 20 	 * @type string
 21 	 */
 22 	separator: System.getProperty("path.separator") || ":",
 23 
 24 	/**
 25 	 * Forward or backward slash depending on which OS the bot is being run.
 26 	 * @type string
 27 	 */
 28 	slash: System.getProperty("file.separator") || "/",
 29 
 30 	/**
 31 	 * The path to the folder where the shell or bat file to start the bot was run.
 32 	 * @type string
 33 	 */
 34 	path: System.getProperty("user.dir") || "./",
 35 
 36 	/**
 37 	 * Includes the given script file into the current runtime. An optional parameter describes whether the path is relative or absolute.
 38 	 * @param {string} path The path to the file.
 39 	 * @param {string} [relative] <code>true</code> if the path is relative to {@link #path}, <code>false</code> if the path is absolute.
 40 	 */
 41 	include: function(path, relative) {
 42 		var error = false;
 43 		if(relative || (relative === undefined)) {
 44 			var fpath = this.path + path;
 45 			if((new java.io.File(fpath)).exists()) {
 46 				load(fpath);
 47 			} else {
 48 				error = fpath;
 49 			}
 50 		} else {
 51 			if((new java.io.File(path)).exists()) {
 52 				load(path);
 53 			} else {
 54 				error = path;
 55 			}
 56 		}
 57 		if(error) {
 58 			var err = new Error("IO.include: File does not exist!");
 59 			err.lineNumber = 0;
 60 			err.fileName = error;
 61 			throw err;
 62 		}
 63 	}
 64 }
 65 
 66 if(IO.path.lastIndexOf(IO.slash) !== IO.path.length) {
 67 	IO.path += IO.slash;
 68 }
 69

 71 IO.include("app/JSON.js");
 72 IO.include("app/util.js");
 73 IO.include("app/timer.js");
 74 IO.include("app/scriptbot.js");
 75 IO.include("app/main.js");
