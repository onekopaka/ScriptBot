1 /**
  2  * Gets contents of the specified URL. An optional parameter determines the format of the returned data.
  3  *
  4  * @param {string} url The url/location to be fetched.
  5  * @param {boolean} [asLines] <code>true</code> returns the contents as a \n (new line) separated string, <code>false</code> to return the contents as an array of lines.
  6  * @returns A string or an Array containing the contents of the requested URL.
  7  */
  8 IO.fetchURL = function(url, asLines) {
  9 	var link = new java.net.URL(url);
 10 	var conn = link.openConnection();
 11 	var reader = new java.io.BufferedReader(new java.io.InputStreamReader(conn.getInputStream()));
 12 	var input = "";
 13 
 14 	var line = "";
 15 	var lines = [];
 16 
 17 	while((input = reader.readLine()) != null) {
 18 		if(asLines) {
 19 			lines.push(input+"");
 20 		} else {
 21 			line += input + "\n";
 22 		}
 23 	}
 24 
 25 	try {
 26 		reader.close();
 27 	} catch(e) {
 28 		// Uhh ... what am I supposed to do with it?
 29 	}
 30 
 31 	return ((asLines)? lines : line);
 32 };
