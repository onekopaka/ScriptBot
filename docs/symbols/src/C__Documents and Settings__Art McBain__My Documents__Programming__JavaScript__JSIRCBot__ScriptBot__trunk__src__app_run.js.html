1 importClass(java.lang.System);

  3 /**
  4  * A collection of items related to helping with input/output, as well as loading other script files.
  5  *
  6  * @namespace
  7  */
  8 var IO = {
  9 	/**
 10 	 * The path separator used by the system.
 11 	 * @type string
 12 	 */
 13 	separator: System.getProperty("path.separator") || ":",
 14 
 15 	/**
 16 	 * Forward or backward slash depending on which OS the bot is being run.
 17 	 * @type string
 18 	 */
 19 	slash: System.getProperty("file.separator") || "/",
 20 
 21 	/**
 22 	 * The path to the folder where the shell or bat file to start the bot was run.
 23 	 * @type string
 24 	 */
 25 	path: System.getProperty("user.dir") || "./",
 26 
 27 	/**
 28 	 * Includes the given script file into the current runtime. An optional parameter describes whether the path is relative or absolute.
 29 	 * @param {string} path The path to the file.
 30 	 * @param {string} [relative] <code>true</code> if the path is relative to {@link #path}, <code>false</code> if the path is absolute.
 31 	 */
 32 	include: function(path, relative) {
 33 		if(relative || (relative === undefined)) {
 34 			load(this.path + path);
 35 		} else {
 36 			load(path);
 37 		}
 38 	}
 39 }
 40
