1 /**********************************************
  2 * A simple time plugin. It responds to "time" *
  3 * and returns the current time of the machine *
  4 * on which the bot is being run.              *
  5 **********************************************/
  6 
  7 if(!this["time"]) {
  8 	time = {};
  9 	time.format = function(value) {
 10 		var a = Number(value);
 11 		return (a + (a > 10 && a < 20 ? "th" : a % 10 === 1 ? "st" : a % 10 === 2 ? "nd" : a % 10 === 3 ? "rd" : "th"));
 12 	};
 13 	time.months = ["January","February","March","April","May","June","July","August","September","October","November","December"];
 14 }
 15 
 16 // Remove any other "time" plugins.
 17 core.unregisterPluginByEvent(Event.MESSAGE, "time");
 18 // Reads off the current time somewhere.
 19 core.registerPlugin(Event.MESSAGE, function(bot, event, args, priv) {
 20 	var msg = args[args.length-1].toLowerCase();
 21 
 22 	// Only respond to "time" messages.
 23 	if(msg === "time" || msg.substring(0, 5) === "time ") {
 24 		msg = Util.trim(msg.substring(4));
 25 		var date = new Date();
 26 
 27 		// Handle bot-local time.
 28 		if(msg.substring(0, 4) === "time") {
 29 			msg = Util.trim(msg.substring(4));
 30 			var hours = date.getHours();
 31 			var minutes = date.getMinutes();
 32 			var post = "";
 33 			if(msg === "12") {
 34 				if(hours > 12) {
 35 					hours = hours - 12;
 36 					post = "pm";
 37 				} else {
 38 					post = "am";
 39 				}
 40 			}
 41 			bot.sendMessage(args[0], "It is now " + hours + ":" + ((minutes < 10)? "0": "") + minutes + ((post)? " " + post : "") + " where I live.");
 42 		} else if(msg === "date") {
 43 			bot.sendMessage(args[0], "Today is " + time.months[date.getMonth()] + " " + time.format(date.getDate()) + ", " + date.getFullYear());
 44 		} else {
 45 			bot.sendMessage(args[0], "It is now " + date);
 46 		}
 47 
 48 		return true;
 49 	}
 50 }, "time");
 51 core.registerPluginInfo("time", function(bot, event, args, priv) {
 52 	bot.sendMessage(args[0], bot.prefix + "time [date|time] [12] - Displays the current time. If \"date\" is supplied, I only return the date, and \"time\" only returns the current time (optionally in 12-hour format).");
 53 });
